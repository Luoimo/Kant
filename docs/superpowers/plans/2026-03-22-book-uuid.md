# Book UUID Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the file-path `book_source` identifier with a stable UUID (`book_id`) throughout the API and orchestrator layer, while keeping internal file storage keyed by human-readable `book_title`.

**Architecture:** `book_id` is generated at ingest time using `uuid.uuid5(NAMESPACE_URL, source_path)` (deterministic — same file = same UUID) and stored in every chunk's ChromaDB metadata. A `resolve_book_by_id()` method on `ChromaStore` maps `book_id → {source, book_title, author}`. All public API endpoints accept `book_id`; `book_source` (file path) and `book_title` are computed values used only internally. Deleting and re-ingesting a book produces the same `book_id`.

**Tech Stack:** Python, FastAPI, ChromaDB, `uuid` stdlib, LangGraph `GraphState`.

def test_ingest_sets_book_id_in_metadata(chroma_store, sample_epub_path):

def resolve_book_by_id(self, book_id: str) -> dict | None:
    """Return {book_id, source, book_title, author} for a given book_id, or None."""
    for entry in self.list_book_titles():

```python
from __future__ import annotations

from pathlib import Path

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel

from backend.config import get_settings
from backend.rag.chroma.chroma_store import ChromaStore
from backend.services.plan_generator import PlanGenerator
from backend.storage.plan_storage import LocalPlanStorage, safe_plan_name

router = APIRouter(prefix="/reader", tags=["reader"])


def _plan_storage() -> LocalPlanStorage:
    return LocalPlanStorage(Path(get_settings().plan_storage_dir))


def _resolve(book_id: str) -> tuple[str, str]:
    """Return (book_title, book_source) for a book_id, raise 404 if not found.

    Note: ChromaStore() is instantiated per request. For a small personal
    library this is acceptable. If latency becomes a concern, replace with
    a module-level singleton.
    """
    entry = ChromaStore().resolve_book_by_id(book_id)
    if not entry:
        raise HTTPException(status_code=404, detail=f"书籍不存在：{book_id}")
    return entry["book_title"], entry["source"]


class ReaderInitRequest(BaseModel):
    reading_goal: str = ""


class ReaderProgressRequest(BaseModel):
    chapter: str


@router.post("/{book_id}/init")
def reader_init(book_id: str, req: ReaderInitRequest) -> dict:
    """Auto-generate a reading plan when user opens a book. Idempotent."""
    book_title, book_source = _resolve(book_id)
    gen = PlanGenerator()
    plan = gen.generate(book_title, book_source=book_source, reading_goal=req.reading_goal)
    return {"book_id": book_id, "book_title": book_title, "plan": plan}


@router.get("/{book_id}/plan")
def reader_get_plan(book_id: str) -> dict:
    """Return the current plan markdown for sidebar display."""
    book_title, _ = _resolve(book_id)
    path = Path(get_settings().plan_storage_dir) / f"{safe_plan_name(book_title)}.md"
    content = path.read_text(encoding="utf-8") if path.exists() else ""
    return {"book_id": book_id, "book_title": book_title, "plan": content}


@router.post("/{book_id}/progress")
def reader_progress(book_id: str, req: ReaderProgressRequest) -> dict:
    """Mark a chapter as complete (toggles checkbox in plan file)."""
    book_title, _ = _resolve(book_id)
    storage = _plan_storage()
    if not storage.find_by_book(book_title):
        raise HTTPException(status_code=404, detail="该书暂无阅读计划，请先初始化")
    if not storage.mark_chapter_done(book_title, req.chapter):
        raise HTTPException(status_code=404, detail=f"未找到章节：{req.chapter}")
    return {"ok": True, "book_id": book_id, "book_title": book_title, "chapter": req.chapter}
```

```python
from __future__ import annotations

from functools import lru_cache
from pathlib import Path

from fastapi import APIRouter, HTTPException
from pydantic import BaseModel

from backend.config import get_settings
from backend.rag.chroma.chroma_store import ChromaStore
from backend.services.note_service import NoteService
from backend.storage.note_vector_store import make_note_vector_store

router = APIRouter(prefix="/notes", tags=["notes"])


@lru_cache(maxsize=1)
def _service() -> NoteService:
    settings = get_settings()
    return NoteService(
        notes_dir=Path(settings.note_storage_dir),
        note_vector_store=make_note_vector_store(settings),
    )


def _resolve_title(book_id: str) -> str:
    """Resolve book_id → book_title, raise 404 if not found.

    Note: ChromaStore() is instantiated per request. For a small personal
    library this is acceptable. If latency becomes a concern, replace with
    a module-level singleton (same pattern as _service()).
    """
    entry = ChromaStore().resolve_book_by_id(book_id)
    if not entry:
        raise HTTPException(status_code=404, detail=f"书籍不存在：{book_id}")
    return entry["book_title"]


class AppendNoteRequest(BaseModel):
    book_id: str
    content: str


@router.get("/books")
def list_note_books() -> list[str]:
    """返回已有笔记的书名列表。"""
    return _service().list_books()


@router.get("/timeline")
def get_timeline(book_id: str | None = None) -> dict:
    """返回结构化元数据供前端时间轴可视化。可按 book_id 过滤。"""
    book_title = _resolve_title(book_id) if book_id else None
    return _service().get_timeline(book_title)


@router.get("/{book_id}")
def get_note(book_id: str) -> dict:
    """返回指定书籍的笔记全文，供前端编辑器展示。"""
    book_title = _resolve_title(book_id)
    content = _service().get_note_content(book_title)
    return {"book_id": book_id, "book_title": book_title, "content": content}


@router.post("/append")
def append_note(req: AppendNoteRequest) -> dict:
    """追加用户手写笔记到指定书籍的笔记文件。"""
    if not req.content.strip():
        raise HTTPException(status_code=400, detail="content 不能为空")
    book_title = _resolve_title(req.book_id)
    _service().append_manual(req.content, book_title)
    return {"ok": True, "book_id": req.book_id, "book_title": book_title}
```
